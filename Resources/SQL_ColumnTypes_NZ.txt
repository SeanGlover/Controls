WITH columnTypes (DSN, DATABASE, OWNER, SCHEMA, TABLE_NAME, OBJECT_TYPE, COLUMN_NBR, COLUMN_NAME, FORMAT, LENGTH, NULLABLE) As (
SELECT CAST('NZSQL' AS CHAR(5))
, T.DATABASE
, T.OWNER
, T.SCHEMA
, T.TABLENAME
, CAST('Table' AS VARCHAR(5))
, C.ATTNUM
, C.ATTNAME
, C.FORMAT_TYPE
, C.ATTLEN
, C.ATTNOTNULL

FROM _v_table T
LEFT JOIN _v_relation_column C
ON (T.OBJID=C.OBJID)

///tableCONDITIONS///

UNION ALL

SELECT CAST('NZSQL' AS CHAR(5))
, V.DATABASE
, V.OWNER
, V.SCHEMA
, V.VIEWNAME
, CAST('View' AS VARCHAR(5))
, C.ATTNUM
, C.ATTNAME
, C.FORMAT_TYPE
, C.ATTLEN
, C.ATTNOTNULL

FROM _v_view V
LEFT JOIN _v_relation_column C
ON (V.OBJID=C.OBJID)

///viewCONDITIONS///

)
--====================
SELECT *
FROM columnTypes
